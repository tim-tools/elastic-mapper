## 1. 快速开始

### Maven 依赖

```xml
<dependency>
    <groupId>io.github.timtools</groupId>
    <artifactId>elastic-mapper-spring-boot-starter</artifactId>
    <version>1.0.0</version>
</dependency>
```

### Spring Boot 配置

```yaml
# application.yml
elastic-mapper:
  hosts:
    - 127.0.0.1:9200
  username: elastic        # 可选
  password: changeme       # 可选
  scheme: http             # 默认 http
  mapper-xml-locations:    # 默认 es-mapper/
    - es-mapper/
```

### 启动类

```java
import io.github.timtools.elasticmapper.spring.scan.ESMapperScan;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@SpringBootApplication
@ESMapperScan("com.example.mapper")   // 扫描 Mapper 接口
public class Application {
    public static void main(String[] args) {
        SpringApplication.run(Application.class, args);
    }
}
```

## 2. Entity 实体定义

```java
import io.github.timtools.elasticmapper.annotations.Id;
import io.github.timtools.elasticmapper.annotations.IndexName;
import io.github.timtools.elasticmapper.annotations.Field;

@IndexName("users")          // ES 索引名（默认: 类名 snake_case）
public class User {

    @Id                         // 文档 _id
    private String id;

    @Field(name = "username")   // ES 字段名（默认: Java 字段名 snake_case）
    private String username;

    @Field(type = "keyword")    // 显式指定 ES 类型
    private String status;

    @Field(type = "text", analyzer = "standard")  // 自定义分词器
    private String bio;

    @Field(type = "keyword", index = false)       // 不索引该字段
    private String remark;

    private Integer age;        // 自动推断: Integer → long

    private Boolean active;     // 自动推断: Boolean → boolean

    @Field(type = "date", format = "yyyy-MM-dd HH:mm:ss")
    private Date createdAt;

    // 必须有无参构造器 + getter/setter
    public User() {}
    // getters / setters ...
}
```

### @Field 属性

| 属性 | 默认值 | 说明 |
|------|--------|------|
| `type` | `""`（自动推断） | ES 字段类型，如 `keyword` / `text` / `long` / `date` / `nested` |
| `name` | `""`（camelCase → snake_case） | ES 字段名 |
| `index` | `true` | 是否对该字段建索引 |
| `analyzer` | `""` | text 字段的自定义分词器，如 `standard` / `ik_max_word` |
| `format` | `""`（用全局 `date-format`） | date 字段的格式化模板 |

### Java → ES 类型映射

| Java 类型 | ES 类型 |
|-----------|---------|
| `String`, `char`, `Character` | `text` |
| `Integer`, `Long`, `Short`, `Byte` | `long` |
| `Float`, `Double`, `BigDecimal` | `double` |
| `Boolean` | `boolean` |
| `Date`, `LocalDate`, `LocalDateTime`, `Instant`, `java.sql.Date`, `java.sql.Timestamp` | `date` |
| `List<T>`, `Collection`, `Iterable`, 数组 | `nested` |
| `Map<String, ?>` | `object` |
| `Enum` | `keyword` |
| 其他 | `object` |

## 3. BaseMapper 内置 CRUD

```java
public interface UserMapper extends BaseMapper<User> {
    // 无需写任何代码，继承即获得以下全部方法
}
```

### Insert

```java
User user = new User("1", "Alice", 28);
ESResponse resp = userMapper.insert(user);
// → PUT /users/_doc/1

List<User> batch = List.of(
    new User("2", "Bob", 35),
    new User("3", "Charlie", 22)
);
ESResponse resp = userMapper.insertBatch(batch);
// → POST /users/_bulk
```

### Select

```java
User user = userMapper.selectById("1");         // GET /users/_doc/1
List<User> all = userMapper.selectList();       // match_all (≤ maxResultWindow)
List<User> found = userMapper.selectByIds(      // POST /users/_mget
    List.of("1", "2", "3"));
```

### Update

```java
// 全量覆盖 (PUT /users/_doc/1) — null 字段会覆盖现有值
user.setAge(29);
userMapper.updateById(user);

// 部分更新 (POST /users/_update/1) — null 字段被跳过
User partial = new User();
partial.setAge(30);
userMapper.updatePartialById(partial);
```

### Delete

```java
userMapper.deleteById("1");                     // DELETE /users/_doc/1
userMapper.deleteByIds(List.of("1", "2"));      // POST /users/_delete_by_query
```

## 4. 注解查询

### @Select — 搜索查询

```java
public interface UserMapper extends BaseMapper<User> {

    @Select("{\"query\":{\"match\":{\"username\":#{keyword}}}}")
    List<User> searchByKeyword(@Param("keyword") String keyword);

    @Select("{\"query\":{\"bool\":{\"filter\":[" +
            "{\"term\":{\"status\":#{status}}}," +
            "{\"range\":{\"age\":{\"gte\":#{minAge}}}}" +
            "]}}}")
    List<User> findByStatusAndMinAge(@Param("status") String status,
                                     @Param("minAge") int minAge);
}
```

### @Update — 按查询更新

```java
@Update("{\"script\":{\"source\":\"ctx._source.status=#{newStatus}\"}," +
        "\"query\":{\"term\":{\"username\":#{username}}}}")
ESResponse updateStatus(@Param("username") String username,
                        @Param("newStatus") String newStatus);
```

### @Delete — 按查询删除

```java
@Delete("{\"query\":{\"term\":{\"status\":#{status}}}}")
ESResponse deleteByStatus(@Param("status") String status);
```

### 占位符语法

| 占位符 | 规则 |
|--------|------|
| `#{param}` | 安全绑定：字符串加引号转义，数字原样输出 |
| `${param}` | 直接替换：不做任何转义，原始字符串拼接 |

## 5. 分页

### from/size 分页（< 10,000 条）

```java
Page<User> page = new Page<>(1, 20);  // 第1页，每页20条
page = userMapper.selectPage(page);

page.getTotal();      // 命中总数
page.getRecords();    // 当前页数据
page.getPages();      // 总页数
page.hasNext();       // 是否有下一页

// 翻到下一页
page = userMapper.selectPage(page.nextPage());
```

### search_after 深分页（> 10,000 条）

```java
Page<User> page = Page.searchAfter(20, "createdAt:desc", "_id:asc");
page = userMapper.selectPage(page);

while (page.hasNext()) {
    page = userMapper.selectPage(page.nextPage());
}
```

## 6. 聚合查询

### 原始访问 (Map)

```java
import com.google.gson.reflect.TypeToken;

AggR<Map<String, Object>> result = userMapper.countByStatus();
Map<String, Object> aggs = result.getAggregations();

result.getTook();          // 查询耗时(ms)
result.getShards();        // 分片信息
result.getHits();          // 命中结果
```

### 类型安全 (@JsonPath)

```java
// 定义聚合结果类型
public class StatusCounts {
    @JsonPath("status_counts.buckets")
    private List<Bucket> buckets;
    // getters/setters...
}

public class Bucket {
    @JsonPath("key")
    private String key;
    @JsonPath("doc_count")
    private Long docCount;
    // getters/setters...
}
```

```java
// Mapper 接口
@Select("{\"size\":0,\"aggs\":{\"status_counts\":{\"terms\":{\"field\":\"status\"}}}}")
AggR<StatusCounts> countByStatus();

// 调用
AggR<StatusCounts> result = userMapper.countByStatus();
result.getAggregations().getBuckets().forEach(b ->
    System.out.println(b.getKey() + ": " + b.getDocCount())
);
```

`@JsonPath` 按 dot-notation 路径从 JSON 树中提取值：

```
@JsonPath("status_counts.buckets")   → json → status_counts → buckets
@JsonPath("key")                     → json → key
@JsonPath("doc_count")               → json → doc_count
```

## 7. XML Mapper

### 文件位置

```
src/main/resources/es-mapper/UserMapper.xml
```

### 基本结构

```xml
<mapper>
    <select id="searchUsers">
        {
            "query": {
                "bool": {
                    <where>
                        <if test="keyword != null and keyword != ''">
                            ,{"must": [{"multi_match": {"query": #{keyword}, "fields": ["username","bio"]}}]}
                        </if>
                        <if test="minAge != null">
                            ,{"filter": [{"range": {"age": {"gte": #{minAge}}}}]}
                        </if>
                    </where>
                }
            },
            "size": 20,
            "sort": [{"createdAt": "desc"}]
        }
    </select>

    <select id="countByStatus">
        {
            "size": 0,
            "aggs": {
                "status_counts": {
                    "terms": {"field": "status", "size": 50}
                }
            }
        }
    </select>
</mapper>
```

### 动态标签

| 标签 | 作用 | 示例 |
|------|------|------|
| `<if test="...">` | 条件包含 | `<if test="name != null">,"name": #{name}</if>` |
| `<foreach>` | 遍历集合 | `<foreach collection="ids" item="id" separator=",">#{id}</foreach>` |
| `<choose>` / `<when>` / `<otherwise>` | 多路分支 | 类似 switch/case |
| `<where>` | 自动去掉开头 `,` 并包裹 `{"bool":{...}}` | — |
| `<set>` | 自动去掉开头 `,` | — |
| `<trim>` | 通用 trim（prefix/suffix/overrides） | `<trim prefix="(" suffix=")" prefixOverrides=",">...</trim>` |

### OGNL 表达式

`test` 属性使用 OGNL 表达式（与 MyBatis 完全一致）：

```
name != null                          // null 检查
name != null and name != ''           // 链式条件
age > 0                               // 数字比较
status == 'active'                    // 字符串相等
!name.isEmpty()                       // 方法调用
list != null and list.size() > 0      // 集合检查
type == 'A' or type == 'B'            // 逻辑运算
```

## 8. 拦截器插件

### 实现 Interceptor

```java
public class SlowQueryLogger implements Interceptor {

    private static final Logger log = LoggerFactory.getLogger(SlowQueryLogger.class);
    private final ThreadLocal<Long> timer = new ThreadLocal<>();

    @Override
    public void before(MethodInvocation invocation) {
        timer.set(System.nanoTime());
    }

    @Override
    public void after(MethodInvocation invocation, Object result) {
        long ms = (System.nanoTime() - timer.get()) / 1_000_000;
        timer.remove();
        if (ms > 500) {
            log.warn("Slow query: {} → {}ms", invocation.getMethod().getName(), ms);
        }
    }
}
```

### 注册拦截器

`InterceptorChain` 是由 Spring Boot Starter 自动配置的一个 Bean。拦截器**不会**被自动发现——你需要将其显式加入链中：

```java
@Configuration
public class InterceptorConfig {

    @Bean
    public SlowQueryLogger slowQueryLogger() {
        return new SlowQueryLogger();
    }

    @Autowired
    public void registerInterceptors(InterceptorChain chain, SlowQueryLogger logger) {
        chain.addInterceptor(logger);
    }
}
```

`before()` 按注册顺序执行，`after()` 按逆序执行。单个拦截器抛出的异常会被捕获并记录日志，不会中断链路或主流程。也可在运行时调用 `chain.removeInterceptor(...)` / `chain.clear()` 动态调整。

## 9. 配置参考

```yaml
elastic-mapper:
  hosts:                          # ES 地址列表 (必填)
    - 127.0.0.1:9200
  scheme: http                    # 协议 (默认 http)
  username: elastic               # Basic Auth 用户名
  password: changeme              # Basic Auth 密码
  connect-timeout-ms: 5000        # 连接超时 (ms)
  socket-timeout-ms: 30000        # Socket 超时 (ms)
  max-conn-total: 30              # 最大连接数
  max-conn-per-route: 10          # 单路由最大连接数
  max-result-window: 10000        # ES max_result_window
  date-format: "yyyy-MM-dd HH:mm:ss" # 默认日期格式
  log-dsl: false                  # 打印 ES 请求 DSL 到日志
  mapper-xml-locations:           # XML 扫描路径
    - es-mapper/
  entity-packages:                # 实体扫描包
    - com.example.entity
```

| 属性 | 默认值 | 说明 |
|------|--------|------|
| `hosts` | — | **必填**，格式 `host:port` |
| `scheme` | `http` | `http` 或 `https` |
| `max-result-window` | `10000` | ES 索引的 `max_result_window` 设置 |
| `log-dsl` | `false` | 开启后每次查询会打印完整 ES JSON 请求体 |
| `mapper-xml-locations` | `["es-mapper/"]` | 支持 Ant 通配符 (`**/*.xml`) |
